integer nLast = 100;
integer nCurrent = 0;
integer nPrev = 0;
while (nCurrent <= nLast) {
System.debug(nCurrent);
    if(nCurrent != 0) {
integer temp = nCurrent;
nCurrent = nCurrent + nPrev;
nPrev = temp;
}else {
nCurrent = nCurrent+1;
}
}
